Syntax highlighting code blocks is simple if using \texttt{minted}, however requires eternal python compilation. This makes very pretty code blocks that are ulikely to compile without fidling.
\ppart
\begin{minted}{python}
def fib(n):
    if n <= 1:
        return n
    else:
        return fib(n-1) + fib(n-2)
\end{minted}
\ppart
\begin{minted}{racket}
(define (fib n)
    (cond [(<= n 1) n]
          [else (+ (fib (- n 1)) (fib (- n 2)))]))
\end{minted}
